<script>
let pages = document.querySelectorAll(".page");
let current = -1;

/* countdown */
let count = 5;
let text = document.getElementById("countText");

function startCountdown(){
    let t = setInterval(()=>{
        text.innerText = count;
        count--;
        if(count < 0){
            clearInterval(t);
            document.getElementById("countdown").classList.remove("active");
            document.getElementById("cakeScreen").classList.add("active");
        }
    },1000);
}

/* cake cut */
function cutCake(){
    document.getElementById("cakeScreen").classList.remove("active");
    showPage(0);
    document.getElementById("music").play();
    fireworks();
    hearts();
    balloons();
}

/* pages */
function showPage(i){
    pages.forEach(p=>p.classList.remove("active"));
    pages[i].classList.add("active");
    current = i;
}

function nextPage(){
    if(current < pages.length-1) showPage(current+1);
}

function prevPage(){
    if(current > 0) showPage(current-1);
}

/* hearts */
function hearts(){
    setInterval(()=>{
        let h = document.createElement("div");
        h.className="heart";
        h.innerHTML="❤️";
        h.style.left=Math.random()*100+"vw";
        document.body.appendChild(h);
        setTimeout(()=>h.remove(),6000);
    },400);
}

/* balloons */
function balloons(){
    setInterval(()=>{
        let b = document.createElement("div");
        b.className="balloon";
        b.style.left=Math.random()*100+"vw";
        b.style.background=["#ff69b4","#ff4d6d","#ff85a2"][Math.floor(Math.random()*3)];
        document.body.appendChild(b);
        setTimeout(()=>b.remove(),8000);
    },800);
}

/* fireworks */
function fireworks(){
    setInterval(()=>{
        let f = document.createElement("div");
        f.className="firework";
        f.style.left=Math.random()*100+"vw";
        f.style.top=Math.random()*100+"vh";
        document.body.appendChild(f);
        setTimeout(()=>f.remove(),1000);
    },300);
}

/* start */
window.onload = startCountdown;
</script>
